<body>
  <h2>String Operations</h2>

  <label>Enter a string: </label>
  <input type="text" id="inputString">
  <br><br>

  <button onclick="reverseString()">Reverse String</button>
  <button onclick="replaceCharacter()">Replace Character</button>
  <button onclick="checkPalindrome()">Check Palindrome</button>

  <p id="result"></p>

  <script>
    // i. Reverse the string
    function reverseString() {
      let str = document.getElementById("inputString").value;
      let reversed = str.split("").reverse().join("");
      document.getElementById("result").innerText = "Reversed String: " + reversed;
    }

    // ii. Replace characters of a string
    function replaceCharacter() {
      let str = document.getElementById("inputString").value;
      let oldChar = prompt("Enter character to replace:");
      let newChar = prompt("Enter new character:");
      let replaced = str.split(oldChar).join(newChar);
      document.getElementById("result").innerText = 
        `After replacing '${oldChar}' with '${newChar}': ${replaced}`;
    }

    // iii. Check if string is palindrome
    function checkPalindrome() {
      let str = document.getElementById("inputString").value.toLowerCase().replace(/\s/g, '');
      let reversed = str.split("").reverse().join("");
      if (str === reversed)
        document.getElementById("result").innerText = "✅ The string is a Palindrome!";
      else
        document.getElementById("result").innerText = "❌ The string is NOT a Palindrome.";
    }
  </script>
