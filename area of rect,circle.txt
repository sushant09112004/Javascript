<body>
  <h2>Area Calculator</h2>

  <h3>Triangle</h3>
  <label>Base: </label><input type="number" id="base"><br>
  <label>Height: </label><input type="number" id="height"><br>
  <button onclick="areaTriangle()">Calculate Area</button>
  <p id="triangleResult"></p>

  <h3>Rectangle</h3>
  <label>Length: </label><input type="number" id="length"><br>
  <label>Width: </label><input type="number" id="width"><br>
  <button onclick="areaRectangle()">Calculate Area</button>
  <p id="rectangleResult"></p>

  <h3>Circle</h3>
  <label>Radius: </label><input type="number" id="radius"><br>
  <button onclick="areaCircle()">Calculate Area</button>
  <p id="circleResult"></p>

  <script>
    // Function to calculate area of triangle
    function areaTriangle() {
      const base = parseFloat(document.getElementById("base").value);
      const height = parseFloat(document.getElementById("height").value);
      const area = 0.5 * base * height;
      document.getElementById("triangleResult").innerText = "Area of Triangle = " + area;
    }

    // Function to calculate area of rectangle
    function areaRectangle() {
      const length = parseFloat(document.getElementById("length").value);
      const width = parseFloat(document.getElementById("width").value);
      const area = length * width;
      document.getElementById("rectangleResult").innerText = "Area of Rectangle = " + area;
    }

    // Function to calculate area of circle
    function areaCircle() {
      const radius = parseFloat(document.getElementById("radius").value);
      const area = Math.PI * radius * radius;
      document.getElementById("circleResult").innerText = "Area of Circle = " + area.toFixed(2);
    }
  </script>
