  <!-- Display -->
  <input type="text" id="display" readonly style="width:200px; text-align:right;"><br><br>

  <!-- Buttons -->
  <div>
    <button onclick="press('7')">7</button>
    <button onclick="press('8')">8</button>
    <button onclick="press('9')">9</button>
    <button onclick="press('/')">/</button><br><br>

    <button onclick="press('4')">4</button>
    <button onclick="press('5')">5</button>
    <button onclick="press('6')">6</button>
    <button onclick="press('*')">*</button><br><br>

    <button onclick="press('1')">1</button>
    <button onclick="press('2')">2</button>
    <button onclick="press('3')">3</button>
    <button onclick="press('-')">-</button><br><br>

    <button onclick="press('0')">0</button>
    <button onclick="press('.')">.</button>
    <button onclick="calculate()">=</button>
    <button onclick="press('+')">+</button><br><br>

    <button onclick="square()">x²</button>
    <button onclick="clearDisplay()">C</button>
  </div>

  <script>
    // Display handling
    function press(value) {
      document.getElementById("display").value += value;
    }

    // Clear input
    function clearDisplay() {
      document.getElementById("display").value = "";
    }

    // Square of a number
    function square() {
      let val = document.getElementById("display").value;
      if (val === "" || isNaN(val)) {
        alert("❌ Please enter a valid number for squaring");
        return;
      }
      document.getElementById("display").value = val * val;
    }

    // Perform calculation
    function calculate() {
      let exp = document.getElementById("display").value;
      try {
        if (exp === "") {
          alert("⚠️ Please enter some values to calculate");
          return;
        }
        if (/[^0-9+\-*/.]/.test(exp)) {
          alert("❌ Invalid input! Use only numbers and operators (+, -, *, /)");
          return;
        }
        let result = eval(exp);
        if (!isFinite(result)) {
          alert("❌ Invalid mathematical operation (e.g., division by zero)");
          return;
        }
        document.getElementById("display").value = result;
      } catch {
        alert("❌ Error in expression");
      }
    }
  </script>
