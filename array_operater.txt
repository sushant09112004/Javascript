<!DOCTYPE html>
<html>
<body>
  <h3>Array Operations</h3>

  <input id="arr" placeholder="Enter elements (comma separated)"> <br><br>
  <input id="rem" placeholder="Element to remove"> <br><br>
  <input id="chk" placeholder="Value to check"> <br><br>
  <button onclick="run()">Show Result</button>

  <p id="out"></p>

  <script>
    function run() {
      let arr = document.getElementById("arr").value.split(",");
      let rem = document.getElementById("rem").value;
      let chk = document.getElementById("chk").value;

      // remove element
      for (let i = 0; i < arr.length; i++) {
        if (arr[i] === rem) {
          arr.splice(i, 1);
          break;
        }
      }

      // check element
      let found = "No";
      for (let i = 0; i < arr.length; i++) {
        if (arr[i] === chk) {
          found = "Yes";
          break;
        }
      }

      // empty array
      arr = [];

      document.getElementById("out").innerHTML =
        "After removing: " + rem + "<br>" +
        "Contains " + chk + "? " + found + "<br>" +
        "Array emptied: " + arr;
    }
  </script>
</body>
</html>
