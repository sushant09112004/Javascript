<!-- Operation Buttons -->
<div>
  <button onclick="chooseOperation('+')">+</button>
  <button onclick="chooseOperation('-')">-</button>
  <button onclick="chooseOperation('*')">*</button>
  <button onclick="calculateFactorial()">!</button>
  <button onclick="calculateResult()">=</button>
  <button onclick="clearDisplay()">C</button>
</div>

<script>
let display = document.getElementById("display");
let firstNumber = null;
let currentOperation = null;

// Append number to display
function appendNumber(num) {
  display.value += num;
}

// Choose operation (+, -, *)
function chooseOperation(op) {
  firstNumber = Number(display.value);
  currentOperation = op;
  display.value = "";
}

// Calculate factorial
function calculateFactorial() {
  let n = Number(display.value);
  let fact = 1;
  for (let i = 1; i <= n; i++) {
    fact *= i;
  }
  display.value = fact;
}

// Calculate result for +, -, *
function calculateResult() {
  let secondNumber = Number(display.value);
  let result = 0;

  if (currentOperation === "+") result = firstNumber + secondNumber;
  else if (currentOperation === "-") result = firstNumber - secondNumber;
  else if (currentOperation === "*") result = firstNumber * secondNumber;

  display.value = result;
}

// Clear display
function clearDisplay() {
  display.value = "";
  firstNumber = null;
  currentOperation = null;
}
</script>
