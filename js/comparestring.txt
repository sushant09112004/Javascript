<h3>String Comparison</h3>

<script>
let str1 = prompt("Enter first string:");
let str2 = prompt("Enter second string:");

// 1Ô∏è‚É£ Compare lengths
document.write("Length of first string: " + str1.length + "<br>");
document.write("Length of second string: " + str2.length + "<br>");
if (str1.length == str2.length)
  document.write("‚úÖ Both strings have the same length.<br><br>");
else
  document.write("‚ùå Strings have different lengths.<br><br>");

// 2Ô∏è‚É£ Compare specific characters
let pos = prompt("Enter position to compare (starting from 0):");
document.write("str1.charAt(" + pos + ") = " + str1.charAt(pos) + "<br>");
document.write("str2.charAt(" + pos + ") = " + str2.charAt(pos) + "<br><br>");

// 3Ô∏è‚É£ Compare last index of a letter
let ch = prompt("Enter a character to find last index:");
document.write("Last index of '" + ch + "' in str1: " + str1.lastIndexOf(ch) + "<br>");
document.write("Last index of '" + ch + "' in str2: " + str2.lastIndexOf(ch) + "<br><br>");

// 4Ô∏è‚É£ Convert both to uppercase
document.write("Uppercase str1: " + str1.toUpperCase() + "<br>");
document.write("Uppercase str2: " + str2.toUpperCase() + "<br><br>");

// 5Ô∏è‚É£ Compare strings alphabetically
let result = str1.localeCompare(str2);
if (result == 0)
  document.write("‚úÖ Both strings are equal alphabetically.<br>");
else if (result < 0)
  document.write("üìò str1 comes before str2 alphabetically.<br>");
else
  document.write("üìï str1 comes after str2 alphabetically.<br>");
</script>
