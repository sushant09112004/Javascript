<h3>String Operations</h3>

<script>
let str = prompt("Enter a string:");

// 1️⃣ Reverse the string
let reversed = str.split("").reverse().join("");
document.write("Reversed String: " + reversed + "<br><br>");

// 2️⃣ Replace a character
let oldChar = prompt("Enter character to replace:");
let newChar = prompt("Enter new character:");
let replaced = str.replaceAll(oldChar, newChar);
document.write("String after replacement: " + replaced + "<br><br>");

// 3️⃣ Check if the string is a palindrome
if (str === reversed)
  document.write("✅ The string is a palindrome.");
else
  document.write("❌ The string is not a palindrome.");
</script>
