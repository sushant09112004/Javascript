<h3>Array Operations</h3>

<script>
let arr = [10, 20, 30, 40, 50];
document.write("Original array: " + arr + "<br><br>");

// 1️⃣ Remove a specific element
let toRemove = Number(prompt("Enter element to remove:"));
let index = arr.indexOf(toRemove);
if (index !== -1) {
  arr.splice(index, 1); // removes 1 element at found index
  document.write("Array after removing " + toRemove + ": " + arr + "<br><br>");
} else {
  document.write(toRemove + " not found in array.<br><br>");
}

// 2️⃣ Check if array contains a specified value
let toCheck = Number(prompt("Enter value to check:"));
if (arr.includes(toCheck)) {
  document.write("✅ Array contains " + toCheck + "<br><br>");
} else {
  document.write("❌ Array does not contain " + toCheck + "<br><br>");
}

// 3️⃣ Empty the array
arr.length = 0;
document.write("Array after emptying: " + arr + "<br>");
</script>
