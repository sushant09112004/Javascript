  <h3>Append Object to Array & Check Type</h3>

  <script>
    // Create an array
    let students = [{ name: "albert", age: 21 }];
    document.write("<b>Original Array:</b> " + JSON.stringify(students) + "<br><br>");

    // Create and append an object
    let newStudent = { name: "john", age: 22 };
    students.push(newStudent);

    document.write("<b>After Appending Object:</b> " + JSON.stringify(students) + "<br><br>");

    // Check if 'students' is an array
    if (Array.isArray(students))
      document.write("<b>Yes ✅</b> 'students' is an array");
    else
      document.write("<b>No ❌</b> it is not an array");
  </script>
