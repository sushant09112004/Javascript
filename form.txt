<!DOCTYPE html>
<html>
<head>
  <title>Student Information Form</title>
</head>
<body>
  <h2>Student Information Form</h2>

  <form onsubmit="return validateForm()">
    Name: <input type="text" id="name"><br><br>
    Address: <input type="text" id="address"><br><br>
    City: <input type="text" id="city"><br><br>
    State: <input type="text" id="state"><br><br>

    Gender:
    <input type="radio" name="gender" value="Male">Male
    <input type="radio" name="gender" value="Female">Female<br><br>

    Mobile Number: <input type="text" id="mobile"><br><br>
    Email ID: <input type="text" id="email"><br><br>

    <input type="submit" value="Submit">
  </form>

  <p id="result"></p>

  <script>
    function validateForm() {
      let name = document.getElementById("name").value.trim();
      let address = document.getElementById("address").value.trim();
      let city = document.getElementById("city").value.trim();
      let state = document.getElementById("state").value.trim();
      let mobile = document.getElementById("mobile").value.trim();
      let email = document.getElementById("email").value.trim();
      let gender = document.querySelector('input[name="gender"]:checked');

      // Check if all fields filled
      if (!name || !address || !city || !state || !mobile || !email || !gender) {
        alert("⚠️ Please fill all fields!");
        return false;
      }

      // Name validation (only alphabets)
      let namePattern = /^[A-Za-z ]+$/;
      if (!namePattern.test(name)) {
        alert("❌ Enter a valid name (only letters allowed)");
        return false;
      }

      // Mobile validation (10 digits only)
      let mobilePattern = /^[0-9]{10}$/;
      if (!mobilePattern.test(mobile)) {
        alert("❌ Enter a valid 10-digit mobile number");
        return false;
      }

      // Email validation (simple pattern)
      let emailPattern = /^[a-zA-Z0-9._%+-]+@[a-z0-9.-]+\.[a-z]{2,}$/;
      if (!emailPattern.test(email)) {
        alert("❌ Enter a valid email address");
        return false;
      }

      document.getElementById("result").innerHTML = 
        "<b>✅ Form Submitted Successfully!</b><br>" +
        "Name: " + name + "<br>" +
        "Address: " + address + "<br>" +
        "City: " + city + "<br>" +
        "State: " + state + "<br>" +
        "Gender: " + gender.value + "<br>" +
        "Mobile: " + mobile + "<br>" +
        "Email: " + email;
      return false; // prevents actual form submission
    }
  </script>
</body>
</html>
