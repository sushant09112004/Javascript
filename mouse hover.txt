  <p>Move your mouse or click inside the box to see background color changes!</p>

  <input type="text" id="nameBox" placeholder="Click or focus here"
         onfocus="focusColor()" onblur="resetColor()">

  <script>
    function changeColor() {
      document.body.style.backgroundColor = "lightblue";
    }

    function focusColor() {
      document.body.style.backgroundColor = "lightgreen";
    }

    function resetColor() {
      document.body.style.backgroundColor = "white";
    }
  </script>
