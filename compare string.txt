  <h3>Results:</h3>
  <p id="result"></p>

  <script>
    function compareStrings() {
      const str1 = document.getElementById("str1").value;
      const str2 = document.getElementById("str2").value;

      let output = "";

      // Method 1: Simple equality check (case-sensitive)
      if (str1 === str2)
        output += "1️⃣ Strings are exactly equal (case-sensitive).<br>";
      else
        output += "1️⃣ Strings are NOT equal (case-sensitive).<br>";

      // Method 2: Case-insensitive comparison
      if (str1.toLowerCase() === str2.toLowerCase())
        output += "2️⃣ Strings are equal (case-insensitive).<br>";
      else
        output += "2️⃣ Strings are NOT equal (case-insensitive).<br>";

      // Method 3: Lexicographical comparison
      if (str1 < str2)
        output += `3️⃣ '${str1}' comes BEFORE '${str2}' (lexicographically).<br>`;
      else if (str1 > str2)
        output += `3️⃣ '${str1}' comes AFTER '${str2}' (lexicographically).<br>`;
      else
        output += "3️⃣ Strings are lexicographically equal.<br>";

      // Method 4: Checking length
      if (str1.length === str2.length)
        output += "4️⃣ Both strings have the same length.<br>";
      else
        output += `4️⃣ String lengths differ — ${str1.length} vs ${str2.length}.<br>`;

      // Method 5: Partial match check
      if (str1.includes(str2))
        output += `5️⃣ '${str1}' contains '${str2}'.<br>`;
      else if (str2.includes(str1))
        output += `5️⃣ '${str2}' contains '${str1}'.<br>`;
      else
        output += "5️⃣ Neither string contains the other.<br>";

      document.getElementById("result").innerHTML = output;
    }
  </script>
